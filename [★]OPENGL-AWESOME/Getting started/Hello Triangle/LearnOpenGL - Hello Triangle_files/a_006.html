                                                                                                                                      voivodfan Felix Armbruster 2 years ago

                                                                                                                                      A problem with this and all the solutions in the replies is that they trigger too often, if you just check for GLFW_PRESS then the resulting code is going to get called for each frame you have the key held down, which is not what you want. You want it to just fire once on being pressed, and then not again until the key has been first released. An easy way to do this is to have, somewhere, the previous state of that button, and then only fire again if the previous state was GLFW_RELEASE, and the current state is GLFW_PRESS.

                                                                                                                                          glFun voivodfan a year ago


                                                                                                                                          // If we just check GLFW_PRESS, it ends up toggling the mode rapidly as several frames
                                                                                                                                          // could have passed during the press.
                                                                                                                                          //
                                                                                                                                          // First indicating we should change on press and then
                                                                                                                                          // actually changing on key release gives us the intended toggling behaviour
                                                                                                                                          static GLenum fillType {GL_FILL};
                                                                                                                                          static bool changePolygonMode {false};


                                                                                                                                          if (glfwGetKey(window, GLFW_KEY_F1) == GLFW_PRESS) {
                                                                                                                                          changePolygonMode = true;
                                                                                                                                          }


                                                                                                                                          if (glfwGetKey(window, GLFW_KEY_F1) == GLFW_RELEASE) {
                                                                                                                                          if (changePolygonMode) {
                                                                                                                                          switch (fillType) {
                                                                                                                                          case GL_FILL:
                                                                                                                                          {
                                                                                                                                          fillType = GL_LINE;
                                                                                                                                          break;
                                                                                                                                          }
                                                                                                                                          case GL_LINE:
                                                                                                                                          {
                                                                                                                                          fillType = GL_POINT;
                                                                                                                                          break;
                                                                                                                                          }
                                                                                                                                          case GL_POINT:
                                                                                                                                          {
                                                                                                                                          fillType = GL_FILL;
                                                                                                                                          break;
                                                                                                                                          }
                                                                                                                                          }


                                                                                                                                          glPolygonMode(GL_FRONT_AND_BACK, fillType);
                                                                                                                                          changePolygonMode = false;
                                                                                                                                          }
                                                                                                                                          }
